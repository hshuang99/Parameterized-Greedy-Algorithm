\chapter{Chosen Improved Greedy Algorithm}
The primary algorithm integrating these cost metrics and greedy heuristics is detailed below:
\begin{algorithm}
\caption{Chosen Improved Greedy Algorithm}
\footnotesize
\begin{algorithmic}[1]
\Require $mat$, matName, greedy, $p$, occur\_time
\Ensure $d$, $Layers$
\State minm, minm\_size $\leftarrow$ system.float\_info.max
\State SIZE $\leftarrow$ len($mat$)
\State $origin$.copy($mat$)
\State $inv$ $\leftarrow$ inverse($mat$)
\State $L_{r}, L_{c}, Ls_{r}, Ls_{c}, row_{op}, col_{op}$ $\leftarrow$ list()
\State $flag \leftarrow False$

\Function{ModifiedRandomizedGreedy}{greedy, p}
	\State \Return $L_{r}, L_{c}, Ls_{r}, Ls_{c}, mat, row_{op}, col_{op}, depth, flag$
\EndFunction

\If{$flag$}
	\State \Return
\EndIf

\Function{LastCheck}{$L_{r}, L_{c}, Ls_{r}, Ls_{c}$}
	\State \Return $Ls_{r}, Ls_{c}$
\EndFunction

\State $reduce$.copy($origin$)
\State size $\leftarrow$ 0
\Function{GenerateReduceMatrix}{$reduce$, $row_{op}, col_{op}$, size}
	\State \Return reduce, size
\EndFunction

\State $ok \leftarrow True$
\If{$reduce$ != $mat$}
	\State $ok = False$
\EndIf

\If{($d > minm$) OR ($d == minm \land size \geq minm\_size$) OR $!ok$}
	\State \Return
\EndIf

\State $minm \leftarrow d$
\State size\_minm $\leftarrow$ size

\State $per$ $\leftarrow$ list()
\Function{GeneratePermutationList}{SIZE, $mat, per$}
	\State \Return $per$
\EndFunction

\State seq $\leftarrow$ list()
\Function{GenerateSequenceList}{seq, $row_{op}, col_{op}$}
	\State \Return seq
\EndFunction

\State layers $\leftarrow$ list()
\Function{GenerateLayersList}{layers, $Ls_{r}, Ls_{c}$}
	\State \Return layers
\EndFunction

\State correct $\leftarrow$ Verify($origin$, layers, seq, $mat$)
\If{!correct}
	\State \Return
\EndIf
\State \Return $d$, layers
\end{algorithmic}
\end{algorithm}
